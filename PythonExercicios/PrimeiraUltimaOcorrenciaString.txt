#Primeira e última ocorrência de uma String
frase = input('Digite uma frase: ')

frase = frase.strip()
frase = frase.lower()
contador = frase.count('a')
position = frase.find('a')
positionL = frase.rfind('a')

print('A letra A aparece {} vezes!'.format(contador))
print('A primeira letra A está na posição: {}!'.format(position+1))
print('A última letra A está na posição: {}!'.format(positionL+1))
