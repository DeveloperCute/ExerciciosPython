from datetime import date


anoNascimento = int(input('Digite o ano de nascimento: '))

anoAtual = date.today()
idade =  anoAtual.year - anoNascimento


if idade == 18:
  print('Ano do seu alistamento!')
elif idade > 18:
  print('Já passou {} anos de seu alistamento!'.format(idade-18))
else:
  print('Falta {} anos para você se alistar!'.format(18-idade))
